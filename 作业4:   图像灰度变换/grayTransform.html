xp = np.linspace(0, 1, 10)
y1 = np.log10(1+xp)    #像素值取对数
y2 = xp **2            #像素值二次方
y3 = xp ** 3           #像素值三次方     
y4 = np.sqrt(xp)       #像素值开方
y5 = np.arctan(xp)       #像素值
print(y1)
