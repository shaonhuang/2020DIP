# 灰度世界算法 GW（ Gray world Assumption）
def balance(img):
    r = img[:,:,0]
    g = img[:,:,1]
    b = img[:,:,2]
    mr = np.mean(r)
    mg = np.mean(g)
    mb = np.mean(b)
    k = (mr + mg + mb) / 3
    kb = k / mb
    kg = k / mg
    kr = k / mr
    
    R = r * kr
    G = g * kg
    B = b * kb
    balanced = np.dstack((R, G, B))
    return balanced
